# np.where(—É—Å–ª–æ–≤–∏–µ, –∑–Ω–∞—á–µ–Ω–∏–µ_–µ—Å–ª–∏_True, –∑–Ω–∞—á–µ–Ω–∏–µ_–µ—Å–ª–∏_False), –∏—Å–ø–æ–ª—å–∑—É–µ—Ç—Å—è –≤ –ø–∞–Ω–¥–∞—Å –¥–ª—è –∑–∞–º–µ–Ω—ã —Å—Ç–æ–ª–±—Ü–æ–≤ —Ç–∞–∫–∂–µ
```python
import pandas as pd
import numpy as np

# –°–æ–∑–¥–∞–µ–º DataFrame
df = pd.DataFrame({
    "Name": ["Alice", "Bob", "Charlie", "David"],
    "Age": [23, 35, 18, 45]
})

# –î–æ–±–∞–≤–ª—è–µ–º —Å—Ç–æ–ª–±–µ—Ü "Category": –µ—Å–ª–∏ Age >= 30 ‚Üí "–í–∑—Ä–æ—Å–ª—ã–π", –∏–Ω–∞—á–µ "–ú–æ–ª–æ–¥–æ–π"
df["Category"] = np.where(df["Age"] >= 30, "–í–∑—Ä–æ—Å–ª—ã–π", "–ú–æ–ª–æ–¥–æ–π")

# –î–æ–±–∞–≤–ª—è–µ–º —Å—Ç–æ–ª–±–µ—Ü "Group" —Å –Ω–µ—Å–∫–æ–ª—å–∫–∏–º–∏ —É—Å–ª–æ–≤–∏—è–º–∏
df["Group"] = np.where(
    df["Age"] < 20, "–ü–æ–¥—Ä–æ—Å—Ç–æ–∫",
    np.where(df["Age"] < 40, "–ú–æ–ª–æ–¥–æ–π –≤–∑—Ä–æ—Å–ª—ã–π", "–ü–æ–∂–∏–ª–æ–π")
)

print(df)
```
